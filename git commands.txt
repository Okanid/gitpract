git init	 				 - создать папку с гит
git status 	 				 - проверить отслеживаемые файлы
git config --list				 - проверить настройки
git add . 					 - добавить все файлы и папки в каталоге
git add "*.html" 				 - добавить файлы с данным расширением
git rm --cached <file>  			 - удалить из отслеживания

для игнора - создать файл .gitignore и внести в него папку или
файлы напр. app/

git commit -a(все файлы которые были изменены поместить в индекс)
-m(указываем коммiт)"init"(массовое добавление)
git commit -a -m "init"

git commit -m "add files"	   		 - отслеживаем1
git commit -m "commit"		   		 - отслеживаем2
git checkout -- название файла.html		 - для отката 1 файла

git log					         - история коммитов
q						 - выход из журнала логов(большого журнала)
git help log 			   		 - помощь по команде (лог 
						 например)
git log --pretty=format:"%h - %an, %ar : %s"     - компактный вид лога
git log --since=2.weeks   			 - показать коммиты за последние 2 недели
git log -p -2(2 комита)			  	 - показывает последний коммит и последние
						 изменения в нем

git checkout -b new_f    			 - создаем ветку new_f и переключаемся
						 на нее -b
git branch 					 - посмотреть ветки
git branch -v 					 - последние комиты в ветках
git branch new_f				 - создаем ветку но не переключаемся на нее
git checkout new_f2				 - переключаемся на указанную ветку

git remote add origin https.... 		 - добавляем репозиторий с названием
				  		 origin в учетную запись гитхаба
git remote					 - просмотр созданных репозиториев
git push -u origin master  			 - говорим гиту сохранять и отправлять в
						 репозиторий origin все изменения ветки master по умолчанию


пример:
1. git status 					 - проверяем статус (красным - непроиндексированные файлы)
2. git add . 					 - (индексируем все файлы)
3. git commit -m"github"  			 - (делаем коммит и комментарий " ")
4. git push  					 - (отправляем на гитхаб, если было указано -u, если
						 нет, то пишем git push origin master)
4.1 git config --global push.default matching    - (будут проталкиваться 
						 все ветки)
4.2 git config --global push.default simple      - (проталкивать только текущую
						 ветку)
5. git clone http... 				 - копируем репозиторий с данной ссылки/аккаунта гитхаба
6. git fetch					 - (если мы в другой ветке)получаем данные из нашего репозитория на гитхабе(данные будут в папке .git)
7. git pull					 - (если мы в другой ветке)получаем данные из нашего репозитория на гитхабе(данные будут в рабочем проекте)
